class Solution:
    def lengthOfLastWord(self, s: str) -> int:
        ls = len(s)
        slower = -1
        while slower >= -ls and s[slower] == ' ':
            slower -= 1
        faster = slower
        while faster >= -ls and s[faster] != ' ':
            faster -= 1
        return slower - faster
