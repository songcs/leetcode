class Solution:
    def firstMissingPositive(self, nums: List[int]) -> int:
        n = len(nums)
        # if n==0: return 1
        for i in range(n):
            while nums[i]>0 and nums[i] <= n and nums[i] != nums[nums[i]-1]:
                idx = nums[i]
                val = nums[idx-1]
                nums[i] = val
                nums[idx-1] = idx
                # nums[i], nums[nums[i]-1] = nums[nums[i]-1], nums[i] #用这个交换会出错

        for i in range(n):
            if nums[i] != i+1:
                return i+1
        return n+1
        # in case: [], [1]
