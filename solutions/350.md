class Solution:
    def intersect(self, nums1: List[int], nums2: List[int]) -> List[int]:
        hs = {}
        for n in nums1:
            hs.setdefault(n,0)
            hs[n] += 1
        ans = []
        for n in nums2:
            hs.setdefault(n,0)
            if hs[n]>0:
                ans.append(n)
                hs[n] -= 1
        return ans
